1. 使用 v-html 在模板中输出元素的 html 代码​

const app = Vue.createApp({
  data() {
    return {
      content: `<p>这是一段<span style="color: hsl(0, 80%, 70%);">HTML</span>代码</p>`,
    };
  },
});
app.mount("#app");
<div id="app">
  <div>
    <!-- 会进行渲染 -->
    <!-- <article v-html="content" /> -->
    <!-- 会原样输出html代码 -->
    <article>{{ content }}</article>
  </div>
</div>

2. v-once:只渲染一次 html 模板​

const app = Vue.createApp({
  data() {
    return {
      list: [1, 2, 3],
    };
  },
});
app.mount("#app");
<div id="app">
  <div>
    <p v-once>list 初始长度： {{ list.length }}</p>
    <p>list 新长度： {{ list.length }}</p>
    <button @click="list.push(list.length + 1)">添加元素</button>
  </div>
</div>

3. 给 v-bind 和 v-on 设置动态参数​

  • 也可以写成<input type="text" :['place'+'holer']="value" @[event]="handleChange" />,但注意不要有+两边不要有空格
  • 如果<input type="text" :[null]="value" @[event]="handleChange" />
const app = Vue.createApp({
  data() {
    return {
      attr: "placeholder",
      value: "请输入一些字符",
      event: "change",
    };
  },
  methods: {
    handleChange() {
      console.log("input 变化");
    },
  },
});
app.mount("#app");
<div id="app">
  <div>
    <input type="text" :[attr]="value" @[event]="handleChange" />
  </div>
</div>

6. 事件传参与多事件处理函数​

  • @click="handleClick(msg)"

  • @click="handleClick(msg, $event)": $event可以访问并传递原生的 js dom 事件

  • @click="handleClick1(), handlerClick2()":可以传递多个事件处理函数, 用,隔开

const app = Vue.createApp({
  data() {
    return {
      todos: [
        {
          id: 1,
          content: "待办1",
        },
        {
          id: 2,
          content: "待办2",
        },
        {
          id: 3,
          content: "待办3",
        },
      ],
    };
  },
  methods: {
    handleClick(id, e) {
      alert(id + ", " + e.clientX + ", " + e.clientY);
    },
    showContent(content) {
      alert(content);
    },
  },
});
app.mount("#app");
<div id="app">
  <div>
    <ul>
      <li
        v-for="todo in todos"
        :key="todo.id"
        @click="handleClick(todo.id, $event), showContent(todo.content)"
      >
        {{todo.content}}
      </li>
    </ul>
  </div>
</div>

7. 常用事件相关的修饰符​

修饰符说明
prevent阻止默认事件
stop阻止冒泡
capturecapture 实现捕获触发事件的机制 即是给元素添加一个监听器,当元素发生冒泡时,先触发带有该修饰符的元素
once只触发一次事件
self只有点击当前元素本身时才会触发回调
const app = Vue.createApp({
  data() {
    return {};
  },
  methods: {
    handleUlClick() {
      alert(0);
    },
    handleLiClick() {
      alert(1);
    },
  },
});
app.mount("#app");
<div id="app">
  <div>
    <a href="https://gugulh.github.io" @click.prevent="">点击跳转</a>
    <!-- <ul @click="handleUlClick">
          <li @click="handleLiClick">1</li>
          <li>2</li>
        </ul>  -->
    <!-- stop-->
    <ul @click="handleUlClick">
      <li @click.stop="handleLiClick">1</li>
      <li>2</li>
    </ul>
    <!-- capture -->
    <!-- <ul @click.capture="handleUlClick">
          <li @click="handleLiClick">1</li>
          <li>2</li>
        </ul> -->
    <!-- self -->
    <!-- <ul @click.self="handleUlClick">
          <li @click="handleLiClick">1</li>
          <li>2</li>
        </ul> -->
    <!-- once -->
    <!-- <ul @click.once="handleUlClick">
          <li @click="handleLiClick">1</li>
          <li>2</li>
        </ul> -->
  </div>
</div>

8. 常见处理键盘,鼠标事件的修饰符​

  • 键盘修饰符: .enter,.tab,.delete,.esc,.up,.down,.left,.right

  • 常见组合键修饰符: .ctrl,.alt,.shift,.meta(Win 是徽标键,Mac 是 Command 键)

  • 鼠标修饰符: .left,.middle,.right

  • .exact: 允许精确地控制修饰符组合触发的事件

const app = Vue.createApp({
  data() {
    return {
      comments: ["好", "非常好", "赞"],
      comment: "",
    };
  },
  methods: {
    addComment() {
      this.comments.push(this.comment);
      this.comment = "";
    },
  },
});
app.mount("#app");
<div id="app">
  <div>
    <ul @mousedown.right="comments.reverse()">
      <li v-for="content in comments">{{ content }}</li>
    </ul>
    <input type="text" v-model="comment" @keyup.ctrl.enter.exact="addComment" />
  </div>
</div>

9. 表单输入相关的修饰符.lazy .trim .number​

  • .lazy: @input => @change
  • .number: 转换为数字类型
  • .trim: 去除收尾空格
const app = Vue.createApp({
  data() {
    return {
      username: "",
      year: 2000,
    };
  },
});
app.mount("#app");
<div id="app">
  <div>
    <input type="text" v-model.lazy.trim="username" />
    <p>{{ username }}</p>
    <!-- <input type="text" v-model.number="year" /> -->
    <input type="number" v-model="year" />
    <p>{{ typeof year }}</p>
  </div>
</div>

10. 通过 Vue 实例访问和修改应用的配置​

  • 实例可以直接访问 data 属性
  • 也可以通过$data访问
  • 可以访问computed, methods 和 watch
  • 适合传统网站和 Vue 相结合的场景
const app = Vue.createApp({
  data() {
    return {
      msg: "你好！",
      name: "张三",
    };
  },
  computed: {
    greetings() {
      return `${this.msg} ${this.name}`;
    },
  },
  methods: {
    changeName(name) {
      this.name = name;
    },
  },
  watch: {
    name(newName) {
      console.log("watch 监听：" + newName);
    },
  },
});
const vm = app.mount("#app");
<div id="app">
  <div>
    <p>{{ msg }} {{ name }}</p>
    <!-- <p>{{ greetings }}</p> -->
  </div>
</div>

11. Vue 生命周期函数​

  • beforeCreate: 调用app.mount()创建 Vue 实例之后, createApp()中配置项生效之前调用
  • created: 在beforeCreate()之后,createApp()中配置项生效之前调用,这个时候 computed,method 和 watch 已经配置好了
  • beforeMount: 在 created 生命周期执行完成之后,应用还没有挂载到app.mount()指定的 html 元素上,在挂载之前会调用beforeMount()生命周期钩子,这个时候应用还没在页面上渲染出来
  • mounted: mounted()会在应用挂载到app.mount()指定的 html 元素后执行,这个时候应用已经在页面上渲染出来了
  • beforeUpdate: 当应用中 html 模板需要重新渲染时,例如 data 中的属性发生了变化,那么在刷新之前会执行beforeUpdate()生命周期钩子
  • updated: updated()是在数据更新之后,html 重新渲染完成之后调用
  • beforeUnmount: beforeUnmount()会在应用卸载前执行,调用app.unmount()会卸载组件,那么正式卸载前,先执行beforeUnmount()生命周期钩子,这个时候 Vue 还是正常的,可以在里面做一些清理收尾操作
  • unmounted: unmounted()生命周期会在应用正式卸载之后调用,这个时候跟应用有关的事件监听,或者指令绑定都已经卸载了
  • 常用的生命周期钩子
createdmounted
1.最早可以访问 this
2.可以修改 data 属性
3.请求远程数据		DOM 节点已挂载,可以修改和访问 DOM
const app = Vue.createApp({
  data() {
    return {
      posts: [],
    };
  },
  methods: {
    async fetchPosts() {
      // 模拟请求远程数据
      const res = await fetch("./sample.json");
      const retData = await res.json();
      this.posts = retData;
    },
  },
  created() {
    this.fetchPosts();
  },
});
const vm = app.mount("#app");
<div id="app">
  <div>
    <ul>
      <li v-for="post in posts" :key="post.id">{{ post.title }}</li>
    </ul>
  </div>
</div>
